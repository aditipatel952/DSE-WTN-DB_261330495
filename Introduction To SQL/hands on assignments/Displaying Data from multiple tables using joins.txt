1.SELECT empno,
       ename,
       sal,
       dname,
       loc
FROM emp
NATURAL JOIN dept;

2.SELECT job,
       mgr,
       sal,
       comm,
       dname
FROM emp
NATURAL JOIN dept
WHERE job = 'SALESMAN';

3.SELECT e.ename,
       e.job,
       e.deptno,
       d.dname
FROM emp e
JOIN dept d
ON e.deptno = d.deptno
WHERE d.loc = 'DALLAS';

4.SELECT e.last_name AS "Employee",
       e.employee_id AS "Emp#",
       m.last_name AS "Manager",
       m.employee_id AS "Mgr#"
FROM employees e
JOIN employees m
ON e.manager_id = m.employee_id;

5.SELECT e.last_name AS "Employee",
       e.employee_id AS "Emp#",
       m.last_name AS "Manager",
       m.employee_id AS "Mgr#"
FROM employees e
LEFT JOIN employees m
ON e.manager_id = m.employee_id
ORDER BY e.employee_id;

6.SELECT e.last_name,
       e.job_id,
       d.department_name,
       e.salary,
       s.grade
FROM employees e
JOIN departments d
ON e.department_id = d.department_id
JOIN salgrade s
ON e.salary BETWEEN s.losal AND s.hisal;

7.SELECT e.ename,
       d.dname
FROM emp e
RIGHT OUTER JOIN dept d
ON e.deptno = d.deptno;

8.SELECT e.last_name AS employee_name,
       e.hire_date  AS employee_hire_date,
       m.last_name AS manager_name,
       m.hire_date AS manager_hire_date
FROM employees e
JOIN employees m
ON e.manager_id = m.employee_id
WHERE e.hire_date < m.hire_date;

9.SELECT empno,
       ename,
       dname,
       loc
FROM emp
JOIN dept
USING (deptno)
WHERE job = 'CLERK';

10.SELECT e.ename,
       e.sal,
       e.mgr,
       d.dname
FROM emp e
JOIN dept d
ON e.deptno = d.deptno
WHERE e.sal > 2000;

11.SELECT e.empno,
       e.ename,
       e.job,
       e.deptno,
       d.dname,
       d.loc
FROM emp e
LEFT OUTER JOIN dept d
ON e.deptno = d.deptno;

12.SELECT e.ename,
       d.dname
FROM emp e
RIGHT OUTER JOIN dept d
ON e.deptno = d.deptno;

13.SELECT e.empno,
       d.dname,
       d.loc
FROM emp e
FULL OUTER JOIN dept d
ON e.deptno = d.deptno;
